package MinigameRotateYan;

import javax.swing.*;
import java.awt.*;
import java.awt.event.*;

public class ImagePanel extends JPanel {
    private Image img;

    public ImagePanel(String path){
        img = new ImageIcon(getClass().getResource(path)).getImage();
    }
    @Override
    protected void paintComponent(Graphics g){
        super.paintComponent(g);
        g.drawImage(img, 0,0,this);
    }
    public void RotateImage(Graphics g, Image img, int x, int y, int h, int w, double angleDeg){
        Graphics2D imgCopy =  (Graphics2D) g.create();
        imgCopy.rotate(Math.toRadians(90), x + w / 2, y + h / 2);
        imgCopy.drawImage(img, x, y, w, h, this);

        imgCopy.dispose();
    }
    public Image getImg(){
        return img;
    }
}
